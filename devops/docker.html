      # Docker

      # Ubuntu 安装

      # 安装必要依赖

      sudo apt update
sudo apt install -y apt-transport-https ca-certificates curl gnupg-agent software-properties-common

      # 添加 Docker 官方源

      curl -fsSL https://download.docker.com/linux/ubuntu/gpg | sudo apt-key add -
sudo add-apt-repository -y "deb [arch=amd64] https://download.docker.com/linux/ubuntu $(lsb_release -cs) stable"
sudo apt update

      # 安装 Docker 引擎

      sudo apt install -y docker-ce docker-ce-cli containerd.io

      # 添加当前用户到 Docker 组

      默认情况使用 Docker 需要在前面添加 sudo 指令，比较麻烦。将当前登录用户添加到 docker 组可实现免 sudo 运行。操作完毕后可能需要注销账号重新登录。

      sudo usermod -aG docker $USER

      # Docker Machine

      使用 Docker Machine 可以快速创建出一个带有 Docker 环境的最小系统，依赖于 VirtualBox 虚拟机。这是我们日常测试开发最便捷的一种方式。

      # 安装

      首先安装 VirtualBox 虚拟机

      brew install --cask virtualbox

      然后再安装 Docker Machine，可以直接到 release (opens new window) 下载或使用下面的命令：

      curl -L https://github.com/docker/machine/releases/download/v0.16.2/docker-machine-`uname -s`-`uname -m` > docker-machine
chmod u+x docker-machine

      # 创建虚拟机

      使用 create 指令创建虚拟机，这里的 demo 是该机器的识别名。

      ./docker-machine create demo

      # 常用命令

      常规操作

      • 列出虚拟机 ./docker-machine ls
      • 连接 SSH ./docker-machine ssh demo
      • 查看虚拟机环境 ./docker-machine env demo

      运行管理

      • 启动虚拟机 ./docker-machine start demo
      • 重启虚拟机 ./docker-machine restart demo
      • 停止虚拟机 ./docker-machine stop demo
      • 杀死虚拟机 ./docker-machine kill demo
      • 移除虚拟机 ./docker-machine rm demo
      • 查看虚拟机状态 ./docker-machine status demo

      # 安装软件

      Docker Machine 所安装的微型系统是 Tiny Core Linux。该系统不能使用 apt 等包管理器，大部分软件需要自行编译。不过倒是带了一个 tce-load 工具，可通过它安装少部分知名软件。

      tce-load -wi vim
tce-load -wi tree

      # Docker Compose

      Docker Compose 使多个容器的管理变得更加方便。通过 YAML 文件配置应用所需的所有服务，然后使用一个命令即可实现所有服务的创建与启动。Docker Compose 的配置默认读取当前目录的 docker-compose.yml 文件。

      # 安装

      curl -L "https://github.com/docker/compose/releases/download/1.28.4/docker-compose-$(uname -s)-$(uname -m)" -o docker-compose
chmod u+x docker-compose

      # 配置示例

      version: '3'

services:
  web:
    build: ./web
    ports:
      - '80:80'
    networks:
      - backtier

  redis:
    image: redis
    ports: ['6379']
    networks:
      - backtier

networks:
  backtier:

      # 常用命令

      • 创建并启动容器 ./docker-compose up
      • 停止服务 ./docker-compose stop
      • 启动服务 ./docker-compose start
      • 重启服务 ./docker-compose restart
      • 列出容器 ./docker-compose ps
      • 构建服务镜像 ./docker-compose build
      • 停止并删除所有资源 ./docker-compose down
      • 设置服务容器的数量 ./docker-compose scale
      • 在运行中容器执行命令 ./docker-compose exec 服务 命令

      # Hello World

      这里演示如何创建一个 Hello World 程序作为 Docker 入门的最小示例，以下是目录结构：

      .
|── Dockerfile
└── hello.sh

      脚本文件 hello.sh

      #!/bin/sh
echo "Hello World"

      镜像配置文件 Dockerfile

      FROM alpine
COPY hello.sh /root
CMD ["/root/hello.sh"]

      两个文件都准备好后，执行 build 命令构建镜像

      docker build -t hello .

      最后运行该镜像，成功打印 Hello World

      docker run hello

      # Dockerfile

      # FROM

      设置基础镜像

      FROM scratch        # 从零开始
FROM ubuntu         # 基于Ubuntu
FROM ubuntu:21.04   # 基于Ubuntu 21.04

      # LABEL

      设置镜像 Metadata 信息

      LABEL maintainer="me@example.com"
LABEL version="0.1"
LABEL description="This is description"

      # EXPOSE

      声明暴露端口，帮助镜像使用者理解

      EXPOSE 3306

      # WORKDIR

      工作目录，如果目录不存在则自动创建

      WORKDIR /root
WORKDIR test
RUN pwd     #/root/test

      # VOLUME

      数据持久化。将容器目录挂载到宿主机目录，由 Docker 自动管理

      VOLUME /var/www/html

      # COPY

      复制文件到容器指定目录

      WORKDIR /root
COPY test.sh .  #/root/test.sh

      # ADD

      添加文件到容器指定目录。与 COPY 类似，区别是 ADD 支持网络文件和自动解压文件（网络文件不会被解压）。

      WORKDIR /root
ADD test.tar.gz .

      # ENV

      设置环境变量

      ENV PHP_VERSION 7.4
RUN apt install -y php-$PHP_VERSION

      # RUN

      运行命令。每一个 RUN 指令创建一个新的镜像分层，为了避免无用分层应该合并多条 RUN 指令为一条。

      RUN apt update && apt install -y nginx
RUN ["apt", "install", "-y", "nginx"]   # 参数形式

      # CMD

      设置容器启动后默认执行的命令。如果 docker run 时指定了其它命令，CMD 命令被忽略。多条 CMD 指令仅最后一个生效。

      CMD echo "Welcome"

      # ENTRYPOINT

      设置容器启动时运行的命令。即使运行容器时指定了其它命令，也不会被忽略，一定会被执行。

      COPY docker-entrypoint.sh /usr/local/bin
ENTRYPOINT ["docker-entrypoint.sh"]

      # Docker 网络

      # 网络原理

      在 Docker 安装时会在宿主主机注册一个名为 docker0 的网络设备。

      同时 Docker 会为每个运行中的容器在宿主机中添加一个 veth 虚拟网络设备（容器内部也有一个），多个 Docker 容器使用各自的 veth 通过 docker0 与外界通讯。详细参考 网络命名空间

      # 网络驱动

      • bridge：默认，桥接模式
      • host：容器与主机共享网络堆栈
      • none：禁用网络
      • overlay：将多个 Docker 守护程序连接在一起，使集群服务能够相互通信

      # 网络链接

      链接名为 test 的容器。链接后可直接使用 test 作为主机名。

      docker run --link test busybox /bin/sh -c "ping test"

      # 自定义 Bridge

      通过自定义 Bridge 的方式，将多个容器放置在同一网络设备下，也可以达到相互链接的效果。

      # 创建 Bridge
docker network create -d bridge my-bridge
# 运行test1容器
docker run -d --name test1 --network my-bridge nginx
# 运行test2容器，成功ping通test1
docker run --name test2 --network my-bridge busybox /bin/sh -c "ping test1"

      # 镜像

      构建镜像

      docker build -t test .

      拉取镜像

      docker pull ubuntu

      列出镜像

      docker images
docker images -a  #全部

      移除镜像

      docker rmi 28f6e2705743
docker rmi $(docker images -a -q)  #全部

      清理所有未使用的镜像、容器、盘符和网络

      docker system prune -a

      # 容器

      运行容器

      docker run ubuntu
docker run -it ubuntu /bin/bash                       #交互式
docker run -d -e MYSQL_ROOT_PASSWORD=my-secret mysql  #添加环境变量

      停止容器

      docker stop 8c274fcce98a
docker stop $(docker ps -a -q)  #全部

      移除容器

      docker rm 8c274fcce98a
docker rm $(docker ps -a -q)  #全部

      移除所有已结束的容器

      docker rm $(docker ps -a -f status=exited -q)

      # 网络

      列出所有网络设备

      docker network ls

      检查某网络设备信息

      docker network inspect 99eb51aab68c

      创建网络设备

      docker network create -d bridge my-bridge

      连接某容器到网络设备

      docker network connect my-bridge 535de38c6554

      # 映射

      设置与宿主机的端口映射，其中 8080 是宿主机端口

      docker run --name web -d -p 8080:80 nginx

      设置与宿主机的文件映射

      docker run -v $(pwd):/var/www/html -it ubuntu /bin/bash

      # 数据卷

      列出所有数据卷

      docker volume ls

      删除指定数据卷

      docker volume rm my-volume

      检查某数据卷信息

      docker volume inspect my-volume

      绑定并挂载数据卷

      docker run -d -v my-volume:/var/lib/mysql mysql

      # 资源限制

      限制内存占用

      docker run --memory=300M php

      设置 CPU 占用权重

      docker run --name test1 --cpu-shares=5 php   #最高占用33%
docker run --name test2 --cpu-shares=10 php  #最高占用66%
