# Golang 计算大文件 MD5

# 对于小文件

func FileMD5(path string) string {
	file, err := os.Open(path)
	if err != nil {
		return ""
	}
	defer file.Close()

	hash := md5.New()
	if _, err := io.Copy(hash, file); err != nil {
		return ""
	}

	return hex.EncodeToString(hash.Sum(nil))
}

# 大文件分块加载

func FileMD5(path string) string {
	file, err := os.Open(path)
	if err != nil {
		return ""
	}
	defer file.Close()

	hash, buf := md5.New(), make([]byte, 1<<20)
	for {
		n, err := file.Read(buf)
		if n == 0 {
			if err == nil {
				continue
			}
			if err == io.EOF {
				break
			}
			return ""
		}

		io.Copy(hash, bytes.NewReader(buf[:n]))
	}

	return hex.EncodeToString(hash.Sum(nil))
}
