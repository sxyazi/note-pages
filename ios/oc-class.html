            # Objective-C 类

            # 基础使用

            main.m

            #import <Foundation/Foundation.h>
#import "Person.h"

int main(int argc, const char * argv[]) {
    // 类方法
    NSLog(@"%d\n", [Person incrementCount]);

    // 实例变量
    Person *p = [[Person alloc] init];
    p->var = 100;
    NSLog(@"%d\n", p->var);

    // 实例属性
    [p initPersonWithName:@"Lily" andAge:20];
    NSLog(@"name: %@, age: %d\n", p.name, p.age);

    return 0;
}

            Person.h

            // 类变量声明
static int count;

@interface Person : NSObject {
    // 实例变量声明
    @public
    int var;
}

// 实例属性声明
@property int age;
@property NSString *name;

// 类方法声明
+ (int)incrementCount;

// 实例方法声明
- (void)initPersonWithName:(NSString *)name andAge:(int)age;

@end

            Person.m

            #import <Foundation/Foundation.h>
#include "Person.h"

@implementation Person

// 类方法定义
+ (int)incrementCount
{
    return count += 10;
}

// 实例方法定义
- (void)initPersonWithName:(NSString *)name andAge:(int)age
{
    _name = name;
    _age = age;
}

@end

            # 类的本质

            1. 所有「类实例对象」都有一个 isa 属性，指向实例化它的「类对象」；

              • 类实例对象对应上面代码中的 Person *p
              • 类实例对象中保存了 实例属性
              • 类对象保存了 实例方法 ，类对象在内存中有且仅有一份

            2. 由于「类对象」也是对象，它也有一个 isa 并指向它的「元类对象」；

              元类对象保存了 类属性 和 类方法

            3. 由于「元类对象」也是对象，它也有一个 isa 并指向「根元类对象」；

              根元类对象是 NSObject 的实例，它保存了每个类所必要的东西，如 + (instancetype)alloc

            4. 由于「根元类对象」也是对象，它也有一个 isa 并指向「它自己」，它万物的起点和终结。

            # 构造方法

            main.m

            #import <Foundation/Foundation.h>
#import "Person.h"

int main(int argc, const char * argv[]) {
    Person *p1 = [[Person alloc] init];
    NSLog(@"p1.name=%@\n", p1.name);

    Person *p2 = [[Person alloc] initWithName:@"Lily"];
    NSLog(@"p2.name=%@\n", p2.name);

    return 0;
}

            Person.h

            @interface Person : NSObject

@property NSString *name;

- (instancetype)initWithName:(NSString *)name;

@end

            Person.m

            @implementation Person

- (instancetype)init
{
    if (self = [super init]) {
        _name = @"Lily";
    }

    return self;
}

- (instancetype)initWithName:(NSString *)name
{
    if (self = [super init]) {
        _name = name;
    }

    return self;
}

@end

            # 工厂方法

            根据 Apple 的规范，每个类中可以有一个或多个以类名开头的方法，用于创建并初始化类实例。可以用它来实现上面「构造函数」中的相同功能。

            main.m

            #import <Foundation/Foundation.h>
#import "Person.h"

int main(int argc, const char * argv[]) {
    Person *p1 = [Person person];
    NSLog(@"p1.name=%@\n", p1.name);

    Person *p2 = [Person personWithName:@"Lily"];
    NSLog(@"p2.name=%@\n", p2.name);

    return 0;
}

            Person.h

            @interface Person : NSObject

@property NSString *name;

+ (instancetype)person;

+ (instancetype)personWithName:(NSString *)name;

@end

            Person.m

            @implementation Person

+ (instancetype)person
{
    Person *p = [[self alloc] init];
    p.name = @"Lily";

    return p;
}

+ (instancetype)personWithName:(NSString *)name
{
    Person *p = [[self alloc] init];
    p.name = name;

    return p;
}

@end
