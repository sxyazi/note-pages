# 小程序嵌套 WebView 使用高德地图

loadScript 函数参见 加载远程脚本

<template>
  <view>
    <view id="amap-container" class="map" tabindex="0" />

    <input class="keyword" v-model="keyword" placeholder="输入关键词以搜索…" />

    <view class="results" v-if="results.poiList">
      <view class="result-line" v-for="poi in results.poiList.pois" @tap="select(poi)">
        <view class="name">{{ poi.name }}</view>
        <view class="address">{{ poi.pname }}{{ poi.cityname }}{{ poi.adname }}</view>
      </view>
    </view>
  </view>
</template>

<script>
import _ from 'lodash'

export default {
  data() {
    return {
      // 地图对象
      map: null,
      // 定位城市
      city: '',
      // 定位坐标
      location: null,
      // 搜索关键字
      keyword: '',
      // 选择坐标
      selected: null,
      // 搜索结果
      results: {},
    }
  },
  watch: {
    keyword: _.debounce(function(value) {
      AMap.plugin('AMap.PlaceSearch', () => {
        const placeSearch = new AMap.PlaceSearch({
          map: this.map,
          city: this.city,
        })

        placeSearch.search(value, (status, result) => {
          this.results = result
        })
      })
    }, 300),
  },
  methods: {
    initMap() {
      this.map = new AMap.Map('amap-container', {
        buildingAnimation: true,
        enableHighAccuracy: true,
      })

      this.geoLocate()
      this.cityLocate()
    },

    // 选择地址
    select(poi) {
      this.selected = poi
      this.map.setCenter([poi.location.lng, poi.location.lat])
      this.map.setZoom(18)
    },

    // 坐标定位
    geoLocate() {
      AMap.plugin('AMap.Geolocation', () => {
        const geolocation = new AMap.Geolocation({ timeout: 3000 })
        geolocation.getCurrentPosition((status, result) => {
          if (status !== 'complete') {
            return console.error('坐标定位失败:' + result.message)
          }

          AMap.plugin('AMap.Geocoder', () => {
            const geocoder = new AMap.Geocoder()
            geocoder.getAddress(result.position, (status, result) => {
              if (status !== 'complete') {
                return console.error('根据经纬度查询地址失败')
              }

              this.city = result.regeocode.addressComponent.city
            })
          })

          this.location = result.position
          if (!this.selected) {
            this.map.setCenter([result.position.lng, result.position.lat])
            this.map.setZoom(15)
          }
        })
      })
    },

    // 城市定位
    cityLocate() {
      AMap.plugin('AMap.CitySearch', () => {
        const citySearch = new AMap.CitySearch()
        citySearch.getLocalCity((status, result) => {
          if (this.location) {
            return
          }

          if (status !== 'complete') {
            return console.error('城市定位失败:' + result.message)
          }

          if (result && result.city && result.bounds) {
            this.city = result.city

            if (!this.selected) {
              this.map.setBounds(result.bounds)
              this.map.setZoom(11)
            }
          }
        })
      })
    },
  },
  async created() {
    if (window.AMap && window.AMapUI) {
      this.initMap()
    } else {
      await loadScript('https://wx.jieyang.la/static/amap/amap.js')
      await loadScript('https://wx.jieyang.la/static/amap/amap-ui.js')
      this.initMap()
    }
  },
}
</script>
