        # 网络命名空间

        # 命名空间

        列出所有命名空间

        ip netns ls

        添加命名空间

        ip netns add test

        在某命名空间下执行命令

        ip netns exec test ip a
ip netns exec test ip link
ip netns exec test ping 192.168.1.1

        启动命名空间下某网络设备

        ip netns exec test ip link set dev lo up
ip netns exec test ip link set dev veth-test1 up

        设置命名空间下某网络设备 IP 地址

        ip netns exec test ip addr add 192.168.1.1/24 dev veth-test1

        # 虚拟网络设备

        添加一对虚拟网络设备

        ip link add veth-test1 type veth peer name veth-test2

        设置虚拟网络设备命名空间

        ip link set veth-test1 netns test1

        # 建立双向网络连接

        # 添加两个命名空间
ip netns add test1
ip netns add test2

# 添加一对虚拟网络设备
ip link add veth-test1 type veth peer name veth-test2

# 关联虚拟网络设备到两个命名空间
ip link set veth-test1 netns test1
ip link set veth-test2 netns test2

# 启动两个命名空间上的虚拟网络设备
ip netns exec test1 ip link set dev veth-test1 up
ip netns exec test2 ip link set dev veth-test2 up

# 设置两个命名空间IP地址
ip netns exec test1 ip addr add 192.168.1.1/24 dev veth-test1
ip netns exec test2 ip addr add 192.168.1.2/24 dev veth-test2

# 成功互相ping通
ip netns exec test1 ping 192.168.1.2
ip netns exec test2 ping 192.168.1.1
