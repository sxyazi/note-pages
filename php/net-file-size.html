    # 获取网络文件尺寸

    function netFileSize($url)
{
  $url = parse_url($url);
  if ($url['scheme'] == 'http') {
    list($host, $port) = [$url['host'], $url['port'] ?? 80];
  } else if($url['scheme'] == 'https') {
    list($host, $port) = ["ssl://{$url['host']}", $url['port'] ?? 443];
  }

  $fp = fsockopen($host, $port, $errno, $errstr, 30);
  if (!$fp) {
    echo "$errstr ($errno)<br />\n";
    return false;
  }

  $buff = '';
  fwrite($fp, "GET {$url['path']} HTTP/1.1\r\nHost: {$url['host']}\r\nConnection: Close\r\n\r\n");
  while (!feof($fp)) {
    $buff .= fgets($fp, 128);
    if (strpos($buff, "\r\n\r\n") !== false) {
      fclose($fp);
      return preg_match('/Content-Length:\s*(\d+)/', $buff, $matches) ? $matches[1] : 0;
    }
  }
}
