# 使用递归

function data_to_tree(&$items, $top = 0)
{
  $result = [];
  foreach ($items as $item) {
    if ($top == $item['parent_id']) {
      $result[] = $item + ['children' => data_to_tree($items, $item['id'])];
    }
  }

  return $result;
}

# 使用迭代

推荐，用到了 PHP 的引用特性，相比递归执行效率更高。

function data_to_tree($items, $top = 0)
{
  $items = array_combine(array_column($items, 'id'), $items);
  foreach ($items as $item) {
    $items[$item['parent_id']]['children'][] = &$items[$item['id']];
  }

  return $items[$top]['children'] ?? [];
}
