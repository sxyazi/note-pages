                  # macOS 使用

                  # 启用按住重复按键支持

                  defaults write -g ApplePressAndHoldEnabled 0

                  # 允许打开任何来源的应用

                  sudo spctl --master-disable

                  # 日常更新脚本

                  brew update
brew upgrade --greedy
brew upgrade --cask --greedy

brew cleanup
brew cleanup --prune 3

npm upgrade -g

                  # 安装常用字体

                  brew tap homebrew/cask-fonts  # You only need to do this once!
brew install --cask font-dejavu
brew install --cask font-source-code-pro

                  # 重新安装 Command Line Tools

                  sudo rm -rf /Library/Developer/CommandLineTools
xcode-select --install

                  # 建立临时 SSH 隧道

                  ssh -N -D 127.0.0.1:3333 root@server

                  # 编译安装冲浪工具

                  替换下面代码中的 $R$ 为 。

                  brew install pcre autoconf automake libtool openssl@1.1

git clone https://github.com/$R$r/$R$-libev.git
cd $R$-libev

./autogen.sh
./configure --with-openssl=/usr/local/opt/openssl@1.1 --disable-documentation
make -j8

cp src/$$-local /usr/local/bin/$$r-local

                  # 移除 Command+Shift+A 默认快捷键

                  在 System Preferences - Keyboard - Shortcuts 中取消勾选 Search man Page Index in Terminal 选项。
