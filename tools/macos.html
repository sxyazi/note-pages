# macOS 使用

# 启用按住重复按键支持

defaults write -g ApplePressAndHoldEnabled -bool false
defaults write -g InitialKeyRepeat -int 13  # normal minimum is 15 (225 ms)
defaults write -g KeyRepeat -int 1          # normal minimum is 2 (30 ms)

# 编译安装冲浪工具

替换下面代码中的 $R$ 为 。

brew install pcre autoconf automake libtool openssl@1.1

git clone https://github.com/$R$r/$R$-libev.git
cd $R$-libev

./autogen.sh
./configure --with-openssl=/usr/local/opt/openssl@1.1 --disable-documentation
make -j8

cp src/$-local /usr/local/bin/$r-local
