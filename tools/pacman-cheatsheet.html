        # Pacman 常用指令

        # 通常

        指令 描述
        pacman -S <pkg> 安装包
        pacman -Ss <keywords> 查找包
        pacman -Rsc <pkg> 卸载包
        pacman -Syu <pkg> 升级包
        pacman -Syu 升级所有包

        # 查询

        指令 描述
        pacman -Qe 查询已安装的包
        pacman -Ql <pkg> 查询包文件
        pacman -Qii <pkg> 查询包信息
        pacman -Qo <file> 查询文件所属包
        pacman -Qs <query> 关键词查询已安装的包

        # 孤立

        指令 描述
        pacman -Qdt 列出不需要的包
        pacman -Rns $(pacman -Qdtq) 卸载不需要的软件包
