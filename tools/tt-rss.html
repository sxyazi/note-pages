        # 自建 Tiny Tiny RSS

        # 安装

        安装必要依赖

        apt install vim git unzip docker-compose -y

        克隆 tt-rss 源

        git clone https://git.tt-rss.org/fox/ttrss-docker-compose.git ttrss-docker
cd ttrss-docker

        复制 .env-dist 为 .env 并配置环境变量

        SELF_URL_PATH=https://rss.33.gl/tt-rss

        构建并在后台启动容器

        docker-compose up -d --build

        使用 Caddy 代理 tt-rss

        echo "deb [trusted=yes] https://apt.fury.io/caddy/ /" | tee -a /etc/apt/sources.list.d/caddy-fury.list
apt update && apt install caddy

        编辑 Caddy 配置文件

        # /etc/caddy/Caddyfile

rss.33.gl:80 {
  encode gzip
  route {
    reverse_proxy http://127.0.0.1:8280
  }
}

        重新启动 Caddy

        service caddy stop
service caddy start

        # 主题

        查看 tt-rss 的挂载点

        docker volume inspect ttrss-docker_app | grep Mountpoint
# /var/lib/docker/volumes/ttrss-docker_app/_data

        下载主题文件

        cd tt-rss/themes.local
wget https://raw.githubusercontent.com/DIYgod/ttrss-theme-rsshub/master/dist/rsshub.css

        # 插件

        # Fever API

        cd tt-rss/plugins.local
git clone https://github.com/DigitalDJ/tinytinyrss-fever-plugin fever

        # Mercury 全文抓取

        克隆插件源码

        cd tt-rss/plugins.local
git clone https://github.com/HenryQW/mercury_fulltext

        注册服务到 docker-compose.yml 的 services 项中

        mercury: # set Mercury Parser API endpoint to `service.mercury:3000` on TTRSS plugin setting page
  image: wangqiru/mercury-parser-api:latest
  container_name: mercury
  expose:
    - 3000
  restart: unless-stopped

        # OpenCC 繁简转换

        克隆插件源码

        cd tt-rss/plugins.local
git clone https://github.com/HenryQW/ttrss_opencc

        注册服务到 docker-compose.yml 的 services 项中

        opencc: # set OpenCC API endpoint to `service.opencc:3000` on TTRSS plugin setting page
  image: wangqiru/opencc-api-server:latest
  container_name: opencc
  environment:
    - NODE_ENV=production
  expose:
    - 3000
  restart: unless-stopped

        # RSSHub

        下载 docker-compose.yml

        wget https://raw.githubusercontent.com/DIYgod/RSSHub/master/docker-compose.yml
docker-compose up -d

        创建 volume 持久化 Redis 缓存

        docker volume create redis-data

        在 docker-compose.yml 中的 environment 项加入

        DEBUG_INFO: 'false'

        使用 Caddy 反向代理



         
         
         
         





        rss.33.gl:80 {
  encode gzip
  route /rsshub* {
    uri strip_prefix /rsshub
    reverse_proxy http://127.0.0.1:1200
  }
  route {
    reverse_proxy http://127.0.0.1:8280
  }
}
