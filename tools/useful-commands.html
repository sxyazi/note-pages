# Fish function

function rp
  if test -z "$argv[1]"
    return 1
  else
    set escaped (echo "$argv[1]" | sed 's.\*.\\\*.g; s.?.\\\?.g; s.\[.\\\[.g; s.\].\\\].g; s.{.\\\{.g; s.}.\\\}.g; s.(.\\\(.g; s.).\\\).g; s.!.\\\!.g; s.\\\.\\\\\\\.g')
  end

  find . -name '*'"$escaped"'*' -exec bash -c '
    newname=${0/"'"$argv[1]"'"/"'"$argv[2]"'"};
    echo "$0 ==> $newname";
    mv "$0" "$newname";' {} \;
end
