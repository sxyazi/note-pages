            # Normal 模式

            # 常用操作

            • :q 退出
            • :wq/ZZ 保存并退出
            • :% s/java//g 统计单词出现个数
            • :% s/java/python/g 批量替换
            • :1,10 s/java/python/g 批量替换（使用范围）
            • ; 继续搜索该行下一个，, 继续搜索该行上一个

            # 快速移动

            文件级别

            • gg 移动到文件开头
            • G 移动到文件结尾
            • zz 把屏幕置为中间
            • h m l 跳转到屏幕的开头/中间/结尾
            • Ctrl-u 上翻页 Ctrl-f 下翻页
            • Ctrl-o Ctrl-i 跳转到上一个/下一个位置

            行级别

            • 0 移动到行首第一个字符
            • $ 移动到行尾
            • ^ 移动到第一个非空白字符
            • g_ 移动到行尾非空白字符

            单词级别

            • w/W 移动到下一个单词开头
            • e/E 移动到下一个单词结尾
            • b/B 移动到上一个单词开头

            字符级别

            • k 上 j 下 h 左 l 右
            • f{char} 移动到 char 字符上
            • t{char} 移动到 char 前一个字符
            • F{char} 移动到 char 字符上（从后往前）

            # 快速修改

            • r 修改当前字符
            • R 修改当前字符（不断向后替换）
            • s 删除当前字符并进入插入模式
            • S 删除整行字符并进入插入模式
            • cw 从当前位置删除单词并进入插入模式
            • caw 删除当前单词并进入插入模式
            • ciw 删除当前单词并进入插入模式（不包含空格）
            • C 删除到整行结尾并进入插入模式
            • ct{char} 删除直到 char 并进入插入模式

            # 恢复删除

            • u 撤销上个修改
            • x 删除当前字符
            • dd 删除当前行
            • dw 从当前位置删除单词
            • daw 删除当前整个单词
            • diw 删除当前整个单词（不包含空格）
            • dt{char} 删除直到 char

            # 快速搜索

            • / ?：向前/向后搜索
            • n N：查找下一个/上一个
            • * #：匹配上一个/下一个当前词

            # 进入 insert 模式

            • i 在当前字符前进入插入模式
            • a 在当前字符后进入插入模式
            • o 在当前的下一行进入插入模式
            • I 在当前行的开头进入插入模式
            • A 在当前行的结尾进入插入模式
            • O 在当前的上一行进入插入模式
            • gi 跳到上次编辑处并进入插入模式

            # 进入 visual 模式

            • v 进入选择模式
            • v 选择行
            • vw 从当前位置选择单词
            • vaw 选择当前整个单词
            • viw 选择当前整个单词（不包含空格）
            • Ctrl-v 块状选择模式

            # Insert 模式

            # 自动补全

            • Ctrl-n/Ctrl-p 补全单词 下一个/上一个
            • Ctrl-x Ctrl-f 补全文件名
            • Ctrl-x Ctrl-o 补全代码，需安装对应语言插件

            # 快速删除

            • Ctrl-h 删除上一个字符
            • Ctrl-w 删除上一个单词
            • Ctrl-u 删除当前行

            # 返回 normal 模式

            • 使用 Esc 或 Ctrl-c/Ctrl-[ 退出为 normal 模式

            # 多文件开发

            • :vs <Ctrl-W> v 竖分屏
            • :sq <Ctrl-W> s 横分屏
            • <Ctrl-W> w 在多窗口间切换
            • <Ctrl-W> p 切换回上次访问窗口
            • <Ctrl-W> k/<Ctrl-W> K 切换/移动到上边的窗口
            • <Ctrl-W> j/<Ctrl-W> J 切换/移动到下边的窗口
            • <Ctrl-W> h/<Ctrl-W> H 切换/移动到左边的窗口
            • <Ctrl-W> l/<Ctrl-W> L 切换/移动到右边的窗口

            # 更换配色方案

            使用 :colorscheme <name> 更换 Vim 配色。

            如果有从网络上下载到的配色文件，放置于 ~/.vim/colors 目录。

            # Vim 配置

            " System
set nocompatible
set encoding=utf-8
syntax enable
syntax on
filetype on
filetype indent on
filetype plugin on
filetype plugin indent on

set mouse=a
set clipboard=unnamed

" 常用插件
call plug#begin('~/.vim/plugged')

Plug 'w0ng/vim-hybrid'
Plug 'mhinz/vim-startify'
Plug 'preservim/nerdtree'
Plug 'ctrlpvim/ctrlp.vim'
Plug 'Yggdroot/indentLine'
Plug 'majutsushi/tagbar'
Plug 'vim-airline/vim-airline'
Plug 'vim-airline/vim-airline-themes'

call plug#end()

" 常用设置
set wrap
set ruler
set number
set relativenumber
set hlsearch
set incsearch
set smartcase
set ignorecase

set showcmd
set wildmenu
set cursorline
colorscheme hybrid

set foldmethod=indent
set pastetoggle=<F2>

let g:hybrid_custom_term_colors = 1
let g:hybrid_reduced_contrast = 1

" 一些方便的映射
let mapleader=' '
let g:mapleader=' '

" 使用 leader-w 直接保存
noremap <LEADER>w :w<CR>
inoremap <LEADER>w <Esc> :w<CR>

" 使用 jj 进入 normal 模式
inoremap jj <Esc>`^

" 切换 Buffer
nnoremap <silent> [b :bprevious<CR>
nnoremap <silent> [n :bnext<CR>

" Use Ctrl-h/j/k/l switch window
noremap <C-h> <C-w>h
noremap <C-j> <C-w>j
noremap <C-k> <C-w>k
noremap <C-1> <C-w>l

" Sudo to write
cnoremap w!! w !sudo tee % > /dev/null

" JSON 格式化
com! FormatJSON %!python3 -m json.tool

" ctrlp settings
let g:ctrlp_map = '<c-p>'

" NERDTree settings
nnoremap <LEADER>f :NERDTreeFind<CR>
nnoremap <LEADER>, :NERDTreeToggle<CR>
let NERDTreeIgnore= [
                \ '\.git$', '\.hg$', '\.svn$', '\.stversions$', '\.pyc$', '\.pyo$', '\.swp$',
                \ '\.DS_Store$', '\.sass-cache$', '__pycache__$', '\.egg-info$', '\.ropeproject$',
                \ ]

" EasyMotion settings
nmap ss <Plug>(easymotion-s2)

" Tagbar
nnoremap <LEADER>t :TagbarToggle<CR>
